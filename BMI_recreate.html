def add_trend(data, color, label):
    filtered_data = data[(data.index > 0) & (data.values > 0)]
    if len(filtered_data) < 2:
        return label
    log_degree = np.log10(filtered_data.index)
    log_prob = np.log10(filtered_data.values)
    slope, intercept = np.polyfit(log_degree, log_prob, 1)
    power_y = 10 ** (intercept + slope * log_degree)
    plt.plot(filtered_data.index, power_y, color=color, linestyle='-')
    return f"{label} = {slope:.3f}"
